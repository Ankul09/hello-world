pipeline {
agent any
tools {
maven 'Maven 3.3.9'
jdk 'jdk8'
}
stages {
stage ('Initialize') {
steps {
sh '''
echo "PATH = ${PATH}"
echo "M2_HOME = ${M2_HOME}"
echo "Java_Home = ${JAVA_HOME}"
'''
}
}
stage('git clone') {
steps {
echo 'git code cloning '
git 'https://github.com/Ankul09/hello-world.git'
}
}
stage ('Build') {
steps {
sh 'mvn -Dmaven.test.failure.ignore=true install'
}
}
}
}
